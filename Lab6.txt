int compareChar(char s1, char s2)
{
  int i;
  for (i = 0; s1[i] == s2[i]; i++)
    if (s1[i] == '\0')
      return 0;
  return s1[i] - s2[i];
}

Country* sortByName(Country p[],int size){
  int check;
  while(1 == 1)
  {
    check = -1;
    for(int i = 0;i<size - 1;i++)
    {
      if (compareChar(p[i].name,p[i+1].name) > 1)
      {
      Country tmp = p[i];
      p[i] = p[i+1];
      p[i+1] = tmp;
      check = 1;
      }
    }
    if (check == -1) break;
  }
  return p;
}
